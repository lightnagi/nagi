public class Solution {
    public int countSegments(String s) {
        if(s == null || s.length() == 0) return 0;
        int result = 0;
        String[] str = s.split(" ");
        for(String s1 : str){
            if(!s1.equals("")) result++;
        }
        return result;
    }
}// 3ms
